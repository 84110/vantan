for(i = 0; i <255; i += 5){
  fill(i,255,255);
  ellipse(i+10,10,10,10);
}

float brightness = 0;

void setup(){
  size(270,20);
  colorMode(HSB,255);
}

void draw(){
  background(0);
  int i;
  for(i = 0; i <255; i += 5){
    fill(i,255,brightness);
    ellipse(i+10,10,10,10);
  }
  brightness++;
  if(255<brightness){
    brightness=0;
  }
}
