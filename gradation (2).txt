float brightness = 0;

void setup(){
  size(270,40);
  colorMode(HSB,255);
}


void draw(){
  background(0);
  int i;
  for(i = 0; i < 255; i += 5){
    fill(i,250,brightness);
    ellipse(i+10,10,10,10);
  }
  
  for(i = 0; i < 255; i += 5){
  fill(i,150,brightness);
  ellipse(i+10,20,10,10);
  }
  
  for(i = 0; i < 255; i += 5){
    fill(i,50,brightness);
    ellipse(i+10,30,10,10);
  }
  
  brightness++;
  
  if(255<brightness){
    brightness = 0;
  }
}
